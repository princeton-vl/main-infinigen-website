```bash
python -m tools.manage_datagen_jobs --num_scenes 1 --specific_seed 46fa3103 \
-o test_$RANDOM -p full/render_image.num_samples=200  --cleanup none \
--pipeline_configs local_16GB monocular blender_gt --config desert fast_terrain_assets \
--wandb disabled --pipeline_overrides LocalScheduleHandler.use_gpu=True
```
